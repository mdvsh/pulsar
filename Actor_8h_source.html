101 struct ColliderCacheHash {
102  size_t operator()(const std::pair<Actor*, Actor*>& pair) const {
103  auto ptr1 = reinterpret_cast<uintptr_t>(pair.first);
104  auto ptr2 = reinterpret_cast<uintptr_t>(pair.second);
105 
106  // A simple symmetric combination of the two pointer hashes, they're always
107  // minmaxed in order.
108  size_t hash1 = std::hash<uintptr_t>{}(ptr1);
109  size_t hash2 = std::hash<uintptr_t>{}(ptr2);
110 
111  // Combine the two hashes inspired by the 'boost::hash_combine' method
112  return hash1 ^ (hash2 + 0x9e3779b9 + (hash1 << 6) + (hash1 >> 2));
113  }

237  /* Do not use SDL_GetKeyboardState(), as it will not observe the input file.
238  */
239  static void SDL_ConsiderInputFile() {
240  /* Lazy Initialize */
241  if (input_status == NOT_INITIALIZED) {
242  LoadSDLEventsFromInputFile();
243  }
244 
245  if (input_status == INPUT_FILE_PRESENT) {
246  if (frame_to_user_input.find(frame_number) != frame_to_user_input.end()) {
247  while (!frame_to_user_input[frame_number].empty()) {
248  SDL_PushEvent(&(frame_to_user_input[frame_number].front()));
249  frame_to_user_input[frame_number].pop();
250  }
251  }
252  }
253 
254  /* Recording mode (primarily for course staff usage) */
255  if (RECORDING_MODE && input_status != INPUT_FILE_PRESENT &&
256  !_autograder_mode) {
257  /* Peek at incoming events so we may record them before the student code
258  * consumes them. */
259  SDL_PumpEvents();
260  SDL_Event incoming_events[100];
261  int num_events = SDL_PeepEvents(incoming_events, 100, SDL_PEEKEVENT,
262  SDL_FIRSTEVENT, SDL_LASTEVENT);
263 
264  /* Filter for relevant events */
265  std::vector<SDL_Event> relevant_events;
266  for (int i = 0; i < num_events; i++) {
267  Uint32 event_type = incoming_events[i].type;
268  if (event_type == SDL_KEYUP || event_type == SDL_KEYDOWN ||
269  event_type == SDL_MOUSEMOTION ||
270  event_type == SDL_MOUSEBUTTONDOWN ||
271  event_type == SDL_MOUSEBUTTONUP || event_type == SDL_MOUSEWHEEL ||
272  event_type == SDL_QUIT)
273  relevant_events.push_back(incoming_events[i]);
274  }
275 
276  /* If there are any events we care about, write them to file. */
277  if (relevant_events.size() > 0) {
278  if (!recording_file.is_open()) {
279  recording_file.open("recorded_sdl_user_input.txt");
280  }
281 
282  /* Every line begins with a frame number */
283  recording_file << frame_number << ";";
284 
285  /* Each frame may have multiple events we need to package up. */
286  for (int i = 0; i < relevant_events.size(); i++) {
287  /* Every event begins with an event type */
288  Uint32 event_type = relevant_events[i].type;
289  recording_file << event_type << ",";
290 
291  /* Each event could be of multiple types, with differing numbers of
292  * parameters. */
293  if (event_type == SDL_KEYDOWN || event_type == SDL_KEYUP) {
294  SDL_Scancode keycode = relevant_events[i].key.keysym.scancode;
295  recording_file << keycode;
296  } else if (event_type == SDL_MOUSEMOTION) {
297  Sint32 x = relevant_events[i].motion.x;
298  Sint32 y = relevant_events[i].motion.y;
299  recording_file << x << "," << y;
300  } else if (event_type == SDL_MOUSEBUTTONDOWN) {
301  int button_index =
302  static_cast<int>(relevant_events[i].button.button);
303  recording_file << button_index;
304  } else if (event_type == SDL_MOUSEBUTTONUP) {
305  int button_index =
306  static_cast<int>(relevant_events[i].button.button);
307  recording_file << button_index;
308  } else if (event_type == SDL_MOUSEWHEEL) {
309  float scroll_amount = relevant_events[i].wheel.preciseY;
310  recording_file << scroll_amount;
311  }
312 
313  /* Every event ends with a semicolon */
314  recording_file << ";";
315  }
316 
317  /* Every line ends with newline, denoting the end of the frame. */
318  recording_file << std::endl;
319  }
320 
321  // Program will auto-close recording_user_input.txt file when it exits.
322  }
323  }

352  /* The engine will aim for 60fps (16ms per frame) during a normal play
353  * session. */
354  /* If the engine detects it is being autograded, it will run as fast as
355  * possible. */
356  static void SDL_Delay() {
357  if (_autograder_mode) {
358  //::SDL_Delay(1); Don't bother delaying at all. Gotta go fast when
359  //: autograding.
360  } else {
361  Uint32 current_frame_end_timestamp =
362  SDL_GetTicks(); // Record end time of the frame
363  Uint32 current_frame_duration_milliseconds =
364  current_frame_end_timestamp - current_frame_start_timestamp;
365  Uint32 desired_frame_duration_milliseconds = 16;
366 
367  int delay_ticks =
368  std::max(static_cast<int>(desired_frame_duration_milliseconds) -
369  static_cast<int>(current_frame_duration_milliseconds),
370  1);
371 
372  ::SDL_Delay(delay_ticks);
373  }
374 
375  current_frame_start_timestamp =
376  SDL_GetTicks(); // Record start time of the frame
377  }

379  static void LoadSDLEventsFromInputFile() {
380  if (IsAutograderMode())
381  _autograder_mode = true;
382 
383  if (!std::filesystem::exists(USER_INPUT_FILENAME)) {
384  input_status = INPUT_FILE_MISSING;
385  return;
386  }
387 
388  std::ifstream infile(USER_INPUT_FILENAME);
389  std::string line;
390 
391  while (std::getline(infile, line)) {
392  std::istringstream iss(line);
393  std::string eventStr, frameStr;
394 
395  std::getline(iss, frameStr, ';');
396  int frameNumber = std::stoi(frameStr);
397 
398  if (frame_to_user_input.find(frameNumber) == frame_to_user_input.end())
399  frame_to_user_input[frameNumber] = std::queue<SDL_Event>();
400 
401  std::queue<SDL_Event>& new_queue = frame_to_user_input[frameNumber];
402 
403  while (std::getline(iss, eventStr, ';')) {
404  /* Identify event type */
405  std::istringstream eventStream(eventStr);
406  std::string event_type_string;
407  std::getline(eventStream, event_type_string, ',');
408 
409  // Remove any '\r' carriage returns we might find
410  // (may be needed for stoi to work on osx / linux when
411  // sdl_user_input.txt has been authored on windows) (depends on how
412  // student chooses to download the test cases)
413  event_type_string.erase(std::remove(event_type_string.begin(),
414  event_type_string.end(), '\r'),
415  event_type_string.end());
416 
417  if (event_type_string == "")
418  continue;
419 
420  Uint32 event_type = std::stoi(event_type_string);
421  SDL_Event fabricated_sdl_event;
422  fabricated_sdl_event.type = event_type;
423 
424  /* Handle unpacking for different event types */
425 
426  if (event_type == SDL_KEYUP || event_type == SDL_KEYDOWN) {
427  std::string keycode;
428  std::getline(eventStream, keycode, ',');
429  if (keycode == "")
430  continue;
431 
432  fabricated_sdl_event.key.keysym.scancode =
433  static_cast<SDL_Scancode>(std::stoi(keycode));
434  } else if (event_type == SDL_MOUSEMOTION) {
435  std::string x_str;
436  std::getline(eventStream, x_str, ',');
437  std::string y_str;
438  std::getline(eventStream, y_str, ',');
439 
440  if (x_str == "" || y_str == "")
441  continue;
442 
443  fabricated_sdl_event.motion.x = static_cast<Sint32>(std::stoi(x_str));
444  fabricated_sdl_event.motion.y = static_cast<Sint32>(std::stoi(y_str));
445  } else if (event_type == SDL_MOUSEBUTTONDOWN ||
446  event_type == SDL_MOUSEBUTTONUP) {
447  std::string mouse_button_index_str;
448  std::getline(eventStream, mouse_button_index_str, ',');
449 
450  if (mouse_button_index_str == "")
451  continue;
452 
453  fabricated_sdl_event.button.button =
454  static_cast<Uint8>(std::stoi(mouse_button_index_str));
455  } else if (event_type == SDL_MOUSEWHEEL) {
456  std::string mouse_wheel_movement_str;
457  std::getline(eventStream, mouse_wheel_movement_str, ',');
458 
459  if (mouse_wheel_movement_str == "")
460  continue;
461 
462  fabricated_sdl_event.wheel.preciseY =
463  std::stof(mouse_wheel_movement_str);
464  }
465 
466  new_queue.push(fabricated_sdl_event);
467  }
468  }
469 
470  input_status = INPUT_FILE_PRESENT;
471  }
